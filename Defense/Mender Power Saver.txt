# link all buildings in area
Initialize:
	set mender projector1

Start:
	getlink link i
	jump NextLink equal link mender
	sensor health link @health
	sensor maxHealth link @maxHealth
	jump NextLink equal health maxHealth

Damaged:
	set hasDamage true
	jump FinishGroup always

NextLink:
	op add i i 1
	jump Start lessThan i @links

FinishGroup:
	control enabled mender hasDamage
	set i 0
	set hasDamage false
