Initialize:
	set number 234
	set fractionSize 2
	printflush message1

Start:
	op mod fraction number 1
	op floor parsed number
	set divider 1
	
	print "Number: "
	jump DividerFound lessThan parsed 10

GetDivider:
	op mul divider divider 10
	jump GetDivider lessThan divider parsed
	op div divider divider 10

DividerFound:
	set foundDivider divider #debug

Parse:
	op idiv digit parsed divider
	print digit
	op mod parsed parsed divider
	op idiv divider divider 10
	jump Parse greaterThan divider 0

GetFraction:
	jump ParseEnd equal fraction 0
	jump ParseEnd equal fractionSize 0
	print "."
	set parsed fraction
	set i 0

ParseFraction:
	op mul parsed parsed 10
	op floor digit parsed
	print digit
	op add i i 1
	jump ParseEnd greaterThanEq i fractionSize
	op mod parsed parsed 1
	jump ParseFraction notEqual parsed 0


ParseEnd:
	printflush message1
	set status "FINISHED"

End:
	jump End always
