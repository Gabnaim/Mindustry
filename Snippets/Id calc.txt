Initialize:
	set extraBits 4

GetFlag:
	op shl procId @thisx 10
	op add procId procId @thisy
	op shl flag procId extraBits

GetInfoMask:
	op shl infoMask 1 extraBits
	op sub infoMask infoMask 1

Test:
	op add flag flag 7

GetProcIdFromFlag:
	op shr unitProcId flag extraBits

GetInfoFromFlag:
	op and info flag infoMask
