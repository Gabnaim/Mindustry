GetUnitType:
	lookup unit unitType i
	set firstUnit null
	set units 0

BindUnit:
	ubind unitType
	jump NextUnitType equal @unit null
  	jump CheckFinished notEqual firstUnit null
	set firstUnit @unit
	jump BindUnit always
	
CheckFinished:
	op add units units 1
	jump BindUnit notEqual firstUnit @unit

FinishCount:
	print unitType
	print ": "
	print units
	print "\n"

NextUnitType:
	op add i i 1
	jump GetUnitType lessThan i 38
	
Finish:
	set i 0
	printflush message1
