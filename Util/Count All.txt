Initialize:
	set useMemory true
	set maxUnitIndex 37 #Serpulo
	set TOTAL_ADDR 38
	printflush message1
	GetMemory:
		set memory cell1
		jump Start equal useMemory false
		jump GetMemory equal memory null

Start:
	set unitTypes 0
	set unitIndex 0

ProcessUnitType:
	lookup unit unitType unitIndex

	ubind unitType
	jump NextUnitType equal @unit null

	set units 0
	set firstUnit @unit
	jump CountProcess always

	UnitLoop:
		ubind unitType
		jump FinishGroup equal firstUnit @unit
	
	CountProcess:
		sensor dead @unit @dead
		jump UnitLoop equal dead true
		op add units units 1
		jump UnitLoop always

FinishGroup:
	print unitType
	print ": "
	print units
	print "\n"
	write units memory unitIndex
	op add unitTypes unitTypes 1

NextUnitType:
	op add unitIndex unitIndex 1
	jump ProcessUnitType lessThanEq unitIndex maxUnitIndex
	
Finish:
	set totalTypes unitTypes
	write totalTypes memory TOTAL_ADDR
	printflush message1
	jump Start always
