Initialize:
	set switch switch1
	jump Initialize equal switch null

Start:
	sensor active switch @enabled
	jump Start equal active false
	lookup unit unitType i
	ubind unitType
	jump Next equal @unit null
	jump Next equal @unit firstUnit
	jump Process notEqual firstUnit null
	set firstUnit @unit

Process:
	ucontrol flag 0

Next:
	op add i i 1
	set firstUnit null
	jump Start lessThan i 38

Finish:
	set i 0
	jump Start always
